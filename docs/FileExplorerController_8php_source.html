38 public function scan(Request $request)
39 {
40 // Récupération sécurisée des paramètres
41 $disk = $request->input('disk'); // GET ou POST
42 $path = $request->input('path') ?? '/';
43
44 // Vérification du disque avant tout accès
45 if (!$disk || !array_key_exists($disk, config('filesystems.disks'))) {
46 abort(403, 'Disque interdit');
47 }
48
49 // scan filesystem
50 $items = $this->fileExplorer->scanDisk($disk, $path);
51
52 // chemins vidéos
53 $videoPaths = collect($items)
54 ->where('type', 'video')
55 ->pluck('path')
56 ->values();
57
58 // mapping disk → colonne BDD
59 $column = match ($disk) {
60 'external_local' => 'chemin_local',
61 'ftp_pad' => 'URI_NAS_PAD',
62 'ftp_arch' => 'URI_NAS_ARCH',
63 default => null,
64 };
65
66 $medias = collect();
67 if ($column && $videoPaths->isNotEmpty()) {
68 $medias = Media::whereIn($column, $videoPaths)
69 ->get()
70 ->keyBy($column);
71 }
72
73 // enrichissement
74 $items = collect($items)->map(function ($item) use ($medias, $column) {
75 if ($item['type'] === 'video' && $column) {
76 $item['media'] = $medias->get($item['path']);
77 }
78 return $item;
79 });
80
81 // Retour HTML pour ton front (AJAX)
82 return view('explorer.tree-item', [
83 'items' => $items,
84 ])->render();

138 if ($status !== 'done') {

161 $activeMap = [];

163
164 $unifiedList = [];

166 foreach ($rawFiles as $file) {
167 if (($file['type'] ?? '') !== 'video') continue;

169 $filename = $file['name'];

171
172 if (isset($archivedPaths[$path])) {
173 continue;
174 }
175
176 if (isset($activeMap[$filename])) {
177 $job = $activeMap[$filename];
178 $unifiedList[] = [

180 'path' => $path,
