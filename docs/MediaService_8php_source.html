114 public function updateMetadata(
115 int $idMedia,
116 string $profReferent,
117 ?string $promotion,

119 ?string $description,
120 array $roles
121 ): bool {

263 ->andWhere('chemin_local', 'exists', true);

268 ->orWhere('prenom', 'like', "%{$kw}%");
269 });
270
271 // Recherche dans le nom du PROJET (C'est ça qui manque !)
272 $q->orWhereHas('projets', function($sq) use ($kw) {
273 $sq->where('libelle', 'like', "%{$kw}%");
274 });
275 });
276 }
277
278 // Garde quand même les filtres spécifiques "au cas où" (pour les tests)
279 if (!empty($filtres['projet'])) {
280 $query->whereHas('projets', function ($q) use ($filtres) {
281 $q->where('projets.id', $filtres['projet'])
282 ->andWhere('chemin_local', 'exists', true);
283 });
284 }
285
286 if (!empty($filtres['promotion'])) {
287 $query->where('promotion', $filtres['promotion']);

417 public function getTechnicalMetadata(Media $media): ?array
418 {
419 Log::info("getTechnicalMetadata called for media #{$media->id}");
420
421 // Priorité: chemin_local > ARCH > PAD
422 // Si chemin_local existe, extraire les métadonnées du fichier local
423 if ($media->chemin_local) {
424 $localPath = storage_path('app/' . ltrim($media->chemin_local, '/'));
425 if (file_exists($localPath)) {
426 Log::info("getTechnicalMetadata: Using local file = {$localPath}");
427 return $this->extractMetadataFromFile($localPath);
428 }
429 }
430
431 // Sinon fallback FTP
432 $remoteVideoPath = $media->URI_NAS_ARCH ?? $media->URI_NAS_PAD;
433
434 if (!$remoteVideoPath) {
435 Log::warning("getTechnicalMetadata: No video path for media #{$media->id}");
436 return null;
437 }
438
439 Log::info("getTechnicalMetadata: Remote path = {$remoteVideoPath}");
440
441 // Determine FTP disk
442 $ftpDisk = null;
443 if ($media->URI_NAS_ARCH) {
444 $ftpDisk = 'ftp_arch';
445 } elseif ($media->URI_NAS_PAD) {
446 $ftpDisk = 'ftp_pad';
447 }
448
449 if (!$ftpDisk) {
450 Log::warning("getTechnicalMetadata: No FTP disk determined for media #{$media->id}");
451 return null;
452 }
453
454 Log::info("getTechnicalMetadata: Using FTP disk = {$ftpDisk}");
455
456 // Build FTP URL
457 $config = config("filesystems.disks.{$ftpDisk}");
458
459 $ftpUrl = sprintf(
460 'ftp://%s:%s@%s/%s',
461 $config['username'],
462 $config['password'],
463 $config['host'],
464 ltrim($remoteVideoPath, '/')
465 );
466
467 Log::info("getTechnicalMetadata: FTP URL constructed (host: {$config['host']})");
468
469 // Use ffprobe to extract metadata
470 $command = sprintf(
471 'ffprobe -v quiet -print_format json -show_format -show_streams %s 2>&1',
472 escapeshellarg($ftpUrl)
473 );
474
475 Log::info("getTechnicalMetadata: Executing ffprobe command");
476
477 exec($command, $output, $returnVar);
478
479 if ($returnVar !== 0) {
480 Log::error("getTechnicalMetadata: ffprobe failed with return code {$returnVar}, output: " . implode("\n", $output));
481 return null;
482 }
483
484 Log::info("getTechnicalMetadata: ffprobe succeeded");
485
486 $json = implode("\n", $output);
487 $data = json_decode($json, true);
488
489 if (!$data || !isset($data['streams']) || !isset($data['format'])) {
490 Log::error("getTechnicalMetadata: Invalid JSON data or missing streams/format. JSON: " . substr($json, 0, 200));
491 return null;
492 }
493
494 Log::info("getTechnicalMetadata: Successfully parsed ffprobe output");
495
496 // Extract relevant information
497 $videoStream = collect($data['streams'])->firstWhere('codec_type', 'video');
498 $audioStream = collect($data['streams'])->firstWhere('codec_type', 'audio');
499
500 $metadata = [
501 'duree' => isset($data['format']['duration']) ? (float)$data['format']['duration'] : null,
502 'duree_format' => null,
503 'taille_octets' => isset($data['format']['size']) ? (int)$data['format']['size'] : null,
504 'taille_format' => null,
505 'bitrate' => isset($data['format']['bit_rate']) ? (int)$data['format']['bit_rate'] : null,
506 ];
507
508 if ($videoStream) {
509 $metadata['largeur'] = $videoStream['width'] ?? null;
510 $metadata['hauteur'] = $videoStream['height'] ?? null;
511 $metadata['resolution'] = null;
512 $metadata['codec_video'] = $videoStream['codec_name'] ?? null;
513 $metadata['fps'] = null;
514
515 if (isset($videoStream['r_frame_rate'])) {
516 $parts = explode('/', $videoStream['r_frame_rate']);
517 if (count($parts) === 2 && $parts[1] > 0) {
518 $metadata['fps'] = round($parts[0] / $parts[1], 2);
519 }
520 }
521
522 if ($metadata['largeur'] && $metadata['hauteur']) {
523 $metadata['resolution'] = $metadata['largeur'] . 'x' . $metadata['hauteur'];
524 }
525 }
526
527 if ($audioStream) {
528 $metadata['codec_audio'] = $audioStream['codec_name'] ?? null;
529 $metadata['sample_rate'] = $audioStream['sample_rate'] ?? null;
530 $metadata['channels'] = $audioStream['channels'] ?? null;
531 }
532
533 // Format duration as HH:MM:SS
534 if ($metadata['duree']) {
535 $seconds = (int)$metadata['duree'];
536 $hours = floor($seconds / 3600);
537 $minutes = floor(($seconds % 3600) / 60);
538 $secs = $seconds % 60;
539 $metadata['duree_format'] = sprintf('%02d:%02d:%02d', $hours, $minutes, $secs);
540 }
541
542 // Format file size
543 if ($metadata['taille_octets']) {
544 $metadata['taille_format'] = $this->formatFileSize($metadata['taille_octets']);
545 }
546
547 return $metadata;
548 }

553 private function formatFileSize(int $bytes): string
554 {
555 $units = ['B', 'KB', 'MB', 'GB', 'TB'];
556 $power = $bytes > 0 ? floor(log($bytes, 1024)) : 0;
557
558 return round($bytes / pow(1024, $power), 2) . ' ' . $units[$power];
