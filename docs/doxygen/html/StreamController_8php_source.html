117 private function streamMP4($ftpDisk, $videoPath, Request $request)
118 {
119 try {
120 $disk = Storage::disk($ftpDisk);
121
122 if (!$disk->exists($videoPath)) {
123 Log::error("Video file not found on FTP: $videoPath");
124 abort(404, 'Fichier vidéo non trouvé');
125 }
126
127 $size = $disk->size($videoPath);
128 $mimeType = $disk->mimeType($videoPath) ?? 'video/mp4';
129
130 // Support pour les requêtes Range (nécessaire pour la navigation dans la vidéo)
131 $start = 0;
132 $end = $size - 1;
133
134 if ($request->hasHeader('Range')) {
135 $range = $request->header('Range');
136 if (preg_match('/bytes=(\d+)-(\d*)/', $range, $matches)) {
137 $start = intval($matches[1]);
138 if (!empty($matches[2])) {
139 $end = intval($matches[2]);
140 }
141 }
142 }
143
144 $length = $end - $start + 1;
145
146 // Headers pour le streaming
147 $headers = [
148 'Content-Type' => $mimeType,
149 'Content-Length' => $length,
150 'Accept-Ranges' => 'bytes',
151 'Content-Range' => "bytes $start-$end/$size",
152 'Cache-Control' => 'no-cache, no-store, must-revalidate',
153 'Pragma' => 'no-cache',
154 'Expires' => '0',
155 ];
156
157 $status = ($start > 0 || $end < ($size - 1)) ? 206 : 200;
158
159 // Stream le fichier
160 return response()->stream(
161 function () use ($disk, $videoPath, $start, $length) {
162 $stream = $disk->readStream($videoPath);
163
164 if ($start > 0) {
165 fseek($stream, $start);
166 }
167
168 $bufferSize = 1024 * 1024; // 1MB buffer
169 $bytesRead = 0;
170
171 while (!feof($stream) && $bytesRead < $length) {
172 $toRead = min($bufferSize, $length - $bytesRead);
173 echo fread($stream, $toRead);
174 $bytesRead += $toRead;
175 flush();
176 }
177
178 fclose($stream);
179 },
180 $status,
181 $headers
182 );
183
184 } catch (\Symfony\Component\HttpKernel\Exception\HttpException $e) {
185 throw $e;
186 } catch (\Exception $e) {
187 Log::error('Error streaming video: ' . $e->getMessage());
188 abort(500, 'Erreur lors du streaming de la vidéo');
189 }

331 private function streamLocalMediaVideo(Media $media, Request $request)
332 {
333 $localPath = env('LOCAL_VIDEO_PATH', 'storage/app/TestFolder');
334 $cleanPath = str_replace('storage/app/', '', $localPath);
335
336 // Essayer plusieurs formats de nom de fichier possibles
337 $possibleFilenames = [
338 $media->mtd_tech_titre, // Titre technique exact
339 pathinfo($media->mtd_tech_titre, PATHINFO_FILENAME) . '.mp4', // Sans extension + .mp4
340 str_replace(' ', '_', $media->mtd_tech_titre), // Espaces remplacés par underscore
341 ];
342
343 // Si le chemin URI contient un nom de fichier, l'utiliser
344 if ($media->chemin_local || $media->URI_NAS_ARCH || $media->URI_NAS_PAD) {
345 $uriPath = $media->chemin_local ?? $media->URI_NAS_ARCH ?? $media->URI_NAS_PAD;
346 $possibleFilenames[] = basename($uriPath);
347 }
348
349 // Chercher le fichier
350 foreach ($possibleFilenames as $filename) {
351 if (empty($filename)) continue;
352
353 $filePath = $cleanPath . '/' . $filename;
354 if (Storage::disk('app')->exists($filePath)) {
355 Log::info("Streaming local video: $filePath");
356 return $this->streamLocalVideo('storage/app/' . $filePath, $request);
357 }
358 }
359
360 // Si aucun fichier trouvé, chercher n'importe quel .mp4 dans le dossier
361 $files = Storage::disk('app')->files($cleanPath);
362 $mp4Files = array_filter($files, function($file) {
363 return pathinfo($file, PATHINFO_EXTENSION) === 'mp4';
364 });
365
366 if (!empty($mp4Files)) {
367 // Prendre le premier fichier MP4 trouvé
368 $firstMp4 = reset($mp4Files);
369 Log::info("Using first MP4 found: $firstMp4");
370 return $this->streamLocalVideo('storage/app/' . $firstMp4, $request);
371 }
372
373 Log::error("No local video found for media #{$media->id}");
374 abort(404, 'Vidéo locale non trouvée');
