172 public function update(Request $request, string $id)
173 {
174 $media = \App\Models\Media::findOrFail($id);
175
176 $validated = $request->validate([
177 // Basic media fields (URIs are managed elsewhere, not in this form)
178 'mtd_tech_titre' => 'required|string|max:255',
179 'promotion' => 'nullable|string|max:255',
180 'type' => 'nullable|string|max:255',
181 'theme' => 'nullable|string|max:255',
182 'description' => 'nullable|string|max:5000',
183
184 // Foreign keys
185 'professeur_id' => 'nullable|exists:professeurs,id',
186 'projet_ids' => 'nullable|array',
187 'projet_ids.*' => 'exists:projets,id',
188
189 // Participations (array of student-role pairs)
190 'participations' => 'nullable|array',
191 'participations.*.eleve_nom' => 'required|string',
192 'participations.*.role_id' => 'required|exists:roles,id',
193
194 // propriétés personalisées
195 'properties' => 'nullable|array',
196 'properties.*.key' => 'nullable|string|max:255',
197 'properties.*.value' => 'nullable',
198 ]);
199
200 // Sanitize single-line fields: remove newlines
201 foreach (['mtd_tech_titre', 'promotion', 'type', 'theme'] as $field) {
202 if (!empty($validated[$field])) {
203 $validated[$field] = preg_replace('/[\r\n]+/', ' ', trim($validated[$field]));
204 }
205 }
206
207 $properties = collect($request->input('properties', []))
208 ->filter(fn ($item) => !empty($item['key']))
209 ->mapWithKeys(fn ($item) => [
210 trim((string) $item['key']) => $item['value'] ?? null
211 ])
212 ->toArray();
213
214 \DB::beginTransaction();
215 try {
216 // Update only the metadata fields (don't touch URIs - they're managed elsewhere)
217 $media->update([
218 'mtd_tech_titre' => $validated['mtd_tech_titre'],
219 'promotion' => $validated['promotion'] ?? null,
220 'type' => $validated['type'] ?? null,
221 'theme' => $validated['theme'] ?? null,
222 'description' => $validated['description'] ?? null,
223 'professeur_id' => $validated['professeur_id'] ?? null,
224 'properties' => $properties,
225 ]);
226
227 // Sync projets (many-to-many)
228 if (isset($validated['projet_ids'])) {
229 $media->projets()->sync($validated['projet_ids']);
230 } else {
231 $media->projets()->sync([]);
232 }
233
234 // Update participations: delete old, create new
235 // We don't use sync() here cause we have 3 columns
236 $media->participations()->delete();
237 if (!empty($validated['participations'])) {
238 foreach ($validated['participations'] as $participation) {
239 $nomSaisi = trim($participation['eleve_nom']);
240
241 if ($nomSaisi) {
242 // Découpage du nom et prénom
243 $parts = explode(' ', $nomSaisi);
244 $prenom = count($parts) > 1 ? array_pop($parts) : '';
245 $nom = implode(' ', $parts) ?: $prenom;
246
247 // Cherche l'élève ou le crée s'il est nouveau
248 $eleve = \App\Models\Eleve::firstOrCreate(
249 ['nom' => $nom, 'prenom' => $prenom]
250 );
251
252 // Crée le lien dans la table pivot
253 \App\Models\Participation::create([
254 'media_id' => $media->id,
255 'eleve_id' => $eleve->id,
256 'role_id' => $participation['role_id'],
257 ]);
258 }
259 }}
260
261 \DB::commit();
262
263 return redirect()->route('medias.show', $media->id)
264 ->with('success', 'Média modifié avec succès!');
265
266 } catch (\Exception $e) {
267 \DB::rollBack();
268 \Log::error('Error updating media: ' . $e->getMessage());
269
270 return redirect()->back()
271 ->withInput()
