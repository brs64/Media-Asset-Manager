64 public function startJob(Request $request)
65 {
66 $media = Media::find($request->input('id'));

68
69 if ($request->input('action') === 'queue') {

73
74 $filePath = $request->input('path');
75 $disk = $request->input('disk');
76 $workflowId = config('btsplay.process.workflow_id');
77
78 $windowsRoot = ($disk === 'nas_arch') ? env('URI_NAS_ARCH_WIN') : env('URI_NAS_PAD_WIN');
79 $uncInputFile = rtrim(str_replace('/', '\\', $windowsRoot), '\\') . '\\' . ltrim(str_replace('/', '\\', $filePath), '\\');
80 $variableData = ltrim(str_replace('/', '\\', dirname($filePath)), '\\') . '\\';

82 try {

84 $jobId = $response['job_id'] ?? 'unknown';

87 'transcode_status' => 'en_cours',

92 } catch (\Exception $e) {
93 $media->update(['transcode_status' => 'echoue']);

98 public function checkStatus($jobId)
99 {
100 try {
101 $apiData = $this->ffastrans->getJobStatus($jobId);
102 $source = $apiData['source'] ?? 'not_found';
103 $rawState = strtolower($apiData['state'] ?? '');
104
105 $progress = $apiData['progress'] ?? 0;
106 $finished = false;
107 $label = 'En cours';
108
109 // Find the media record to update its status in the DB during polling
110 $media = Media::where('transcode_job_id', $jobId)->first();
111
112 if (in_array($rawState, ['success', 'finished', 'done', 'terminé']) || ($source === 'history' && !in_array($rawState, ['error', 'failed']))) {
113 $label = 'Terminé';
114 $progress = 100;
115 $finished = true;
116 if ($media) $media->update(['transcode_status' => 'termine']);
117 }
118 elseif (in_array($rawState, ['error', 'failed', 'aborted', 'echoué'])) {
119 $label = 'Echoué';
120 $finished = true;
121 if ($media) $media->update(['transcode_status' => 'echoue']);
122 }
123 elseif ($source === 'active') {
124 $label = "Node " . ($apiData['steps'] ?? '?') . ": " . ($apiData['proc'] ?? 'Traitement');
125 $finished = false;
126 if ($media && $media->transcode_status !== 'en_cours') {
127 $media->update(['transcode_status' => 'en_cours']);
128 }
129 }
130
131 return response()->json([
132 'progress' => $progress,
133 'label' => $label,
134 'finished' => $finished
135 ]);
136 } catch (\Exception $e) {
137 return response()->json(['error' => $e->getMessage()], 500);
138 }
