101 public function handle()
102 {
103 Log::info("Scan FTP démarré avec Live Feedback", ['scan_id' => $this->scanId]);
104
105 // We will hold files here temporarily until we have 10 of them
106 $buffer = [];
107
108 try {
109 FileExplorerService::scanDiskRecursive(
110 $this->disk,
111 $this->path,
112 function ($item) use (&$buffer) {
113
114 // We only care about VIDEO files for the live counter
115 if (($item['type'] ?? '') === 'video') {
116 $buffer[] = $item;
117
118 // EVERY 10 VIDEOS -> Update the Cache
119 if (count($buffer) >= 10) {
120 $this->flushBufferToCache($buffer);
121 $buffer = []; // Empty the buffer
122 }
123 }
124 }
125 );
126
127 // Save any remaining files in the buffer (e.g. the last 4 files)
128 if (count($buffer) > 0) {
129 $this->flushBufferToCache($buffer);
130 }
131
132 // Mark as DONE
133 Cache::put("scan:{$this->scanId}:status", 'done', now()->addHours(2));
134 Log::info("Scan FTP terminé avec succès");
135
136 } catch (\Throwable $e) {
137 Log::error("Erreur Scan: " . $e->getMessage());
138 Cache::put("scan:{$this->scanId}:status", 'failed', now()->addHours(2));
139 }
140 }

142 // Helper function to keep code clean
143 protected function flushBufferToCache(array $newFiles)
144 {
145 $cacheKey = "scan:{$this->scanId}:results";
146
147 // 1. Get current list
148 $currentList = Cache::get($cacheKey, []);
149
150 // 2. Merge new files
151 $updatedList = array_merge($currentList, $newFiles);
152
153 // 3. Save back to cache
154 Cache::put($cacheKey, $updatedList, now()->addHours(2));
155
156 // 4. Update the visible counter
157 Cache::put("scan:{$this->scanId}:count", count($updatedList), now()->addHours(2));
158 }
